create schema "Bibliotek";
set search_path = 'Bibliotek';

create domain brugernavn varchar;
create domain password varchar;
create domain brugerID integer;
create domain superBruger boolean;
create domain tekst text;
create domain titel varchar;
create domain forfatter varchar;
create domain bogID integer;
create domain emne varchar;
create domain sprog varchar;
create domain books varchar;

create table bruger(
  brugernavn brugernavn,
  password password,
  brugerID brugerID,
  superBruger superBruger,
  books books,
  primary key (brugerID)
);

create table bog(
  tekst tekst,
  titel titel,
  forfatter forfatter,
  bogID bogID,
  emne emne,
  sprog sprog,
  primary key (bogID)
);
